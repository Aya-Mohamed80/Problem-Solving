#include <bits/stdc++.h>
using namespace std;
void Hard_Compare(long long a , long long b , long long c , long long d)
{
    cout << ((b * log(a) > d * log(c)) ? "YES" : "NO") << endl;
}
int main()
{
    long long a = 0 , b = 0 , c = 0 , d = 0 ;
    cin>> a >> b >> c >> d;
    Hard_Compare(a , b , c , d);
    return 0;
}
