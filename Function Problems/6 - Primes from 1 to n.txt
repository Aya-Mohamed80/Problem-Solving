#include<bits/stdc++.h>
using namespace std;
void Primes(int num)
{
    int coun = 0;
    for (int j = 1; j <= num; j++) 
        {
            coun = 0;
            for (int i = 1; i <= j; i++) 
            {
                if (j % i == 0)
                    coun++; 
            }
            if (coun == 2)
                cout << j << " ";
        }
}
int main()
{
    long long num = 0;
    cin >> num;
    Primes(num);
    return 0;
}
