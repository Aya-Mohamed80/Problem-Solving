#include <bits/stdc++.h>
using namespace std;
void Some_Sums(int n , int a , int b)
{
    int sum=0,digit=0,sumd=0;
     for(int i=1;i<=n;i++)
        {
            digit=i;
            while(digit != 0) 
            {
              sumd += (digit % 10);
              digit /= 10;
            }
            if( sumd >= a && sumd <=b)
                sum += i;
            
            sumd=0;
        }
          cout<<sum;
}

int main()
{
    long long n=0,a=0,b=0;
    cin>>n>>a>>b;
    Some_Sums(n,a,b);
    return 0;
}
