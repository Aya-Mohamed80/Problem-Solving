#include<bits/stdc++.h>
using namespace std;
void FCR(double num1, double num2)
{
    double result = 0.0;
    result = floor(num1 / num2);
    cout << "floor " << num1 << " / " << num2 << " = " << result << endl;
    result = ceil(num1 / num2);
    cout << "ceil " << num1 << " / " << num2 << " = " << result << endl;
    result = round(num1 / num2);
    cout << "round " << num1 << " / " << num2 << " = " << result << endl;
}
int main()
{
    double num1 = 0.0, num2 = 0.0;
    cin >> num1>> num2;
    FCR(num1,num2);
    return 0;
}
